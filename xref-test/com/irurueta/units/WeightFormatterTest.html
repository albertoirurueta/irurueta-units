31      @Test
32      void testConstructor() {
33          // test empty constructor
34          var formatter = new WeightFormatter();
35  
36          // check
37          assertEquals(Locale.getDefault(), formatter.getLocale());
38          assertEquals(NumberFormat.getInstance().getMaximumFractionDigits(), formatter.getMaximumFractionDigits());
39          assertEquals(NumberFormat.getInstance().getMaximumIntegerDigits(), formatter.getMaximumIntegerDigits());
40          assertEquals(NumberFormat.getInstance().getMinimumFractionDigits(), formatter.getMinimumFractionDigits());
41          assertEquals(NumberFormat.getInstance().getMinimumIntegerDigits(), formatter.getMinimumIntegerDigits());
42          assertEquals(NumberFormat.getInstance().getRoundingMode(), formatter.getRoundingMode());
43          assertEquals(UnitLocale.getDefault(), formatter.getUnitSystem());
44          assertEquals(NumberFormat.getInstance().isGroupingUsed(), formatter.isGroupingUsed());
45          assertEquals(NumberFormat.getInstance().isParseIntegerOnly(), formatter.isParseIntegerOnly());
46          assertEquals(MeasureFormatter.DEFAULT_VALUE_AND_UNIT_FORMAT_PATTERN, formatter.getValueAndUnitFormatPattern());
47  
48          // test constructor with locale
49          final var locale = new Locale("es", "ES");
50          formatter = new WeightFormatter(locale);
51  
52          // check
53          assertEquals(locale, formatter.getLocale());
54          assertEquals(NumberFormat.getInstance(locale).getMaximumFractionDigits(), formatter.getMaximumFractionDigits());
55          assertEquals(NumberFormat.getInstance(locale).getMaximumIntegerDigits(), formatter.getMaximumIntegerDigits());
56          assertEquals(NumberFormat.getInstance(locale).getMinimumFractionDigits(), formatter.getMinimumFractionDigits());
57          assertEquals(NumberFormat.getInstance(locale).getMinimumIntegerDigits(), formatter.getMinimumIntegerDigits());
58          assertEquals(NumberFormat.getInstance(locale).getRoundingMode(), formatter.getRoundingMode());
59          assertEquals(UnitLocale.getFrom(locale), formatter.getUnitSystem());
60          assertEquals(NumberFormat.getInstance(locale).isGroupingUsed(), formatter.isGroupingUsed());
61          assertEquals(NumberFormat.getInstance(locale).isParseIntegerOnly(), formatter.isParseIntegerOnly());
62          assertEquals(MeasureFormatter.DEFAULT_VALUE_AND_UNIT_FORMAT_PATTERN, formatter.getValueAndUnitFormatPattern());
63  
64  
65          // force IllegalArgumentException
66          assertThrows(IllegalArgumentException.class, () -> new WeightFormatter((Locale) null));
67  
68          // test copy constructor
69          formatter = new WeightFormatter(locale);
70          final var formatter2 = new WeightFormatter(formatter);
71  
72          // check
73          assertEquals(locale, formatter2.getLocale());
74          assertEquals(NumberFormat.getInstance(locale).getMaximumFractionDigits(),
75                  formatter2.getMaximumFractionDigits());
76          assertEquals(NumberFormat.getInstance(locale).getMaximumIntegerDigits(), formatter2.getMaximumIntegerDigits());
77          assertEquals(NumberFormat.getInstance(locale).getMinimumFractionDigits(),
78                  formatter2.getMinimumFractionDigits());
79          assertEquals(NumberFormat.getInstance(locale).getMinimumIntegerDigits(), formatter2.getMinimumIntegerDigits());
80          assertEquals(NumberFormat.getInstance(locale).getRoundingMode(), formatter2.getRoundingMode());
81          assertEquals(UnitLocale.getFrom(locale), formatter2.getUnitSystem());
82          assertEquals(NumberFormat.getInstance(locale).isGroupingUsed(), formatter2.isGroupingUsed());
83          assertEquals(NumberFormat.getInstance(locale).isParseIntegerOnly(), formatter2.isParseIntegerOnly());
84          assertEquals(MeasureFormatter.DEFAULT_VALUE_AND_UNIT_FORMAT_PATTERN, formatter2.getValueAndUnitFormatPattern());
85  
86          // Force NullPointerException
87          //noinspection DataFlowIssue
88          assertThrows(NullPointerException.class, () -> new WeightFormatter((WeightFormatter) null));
89      }
