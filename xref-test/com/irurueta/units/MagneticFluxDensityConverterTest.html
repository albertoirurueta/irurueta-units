383     @Test
384     public void testConvertMagneticFluxDensity() {
385         final double value = new Random().nextDouble();
386         final MagneticFluxDensity inputB = new MagneticFluxDensity(
387                 value, MagneticFluxDensityUnit.MICROTESLA);
388 
389         final MagneticFluxDensity outputB = new MagneticFluxDensity();
390         MagneticFluxDensityConverter.convert(inputB, MagneticFluxDensityUnit.TESLA,
391                 outputB);
392 
393         // check
394         assertEquals(inputB.getValue().doubleValue(), value, 0.0);
395         assertEquals(inputB.getUnit(), MagneticFluxDensityUnit.MICROTESLA);
396 
397         assertEquals(outputB.getUnit(), MagneticFluxDensityUnit.TESLA);
398         assertEquals(outputB.getValue().doubleValue(),
399                 MagneticFluxDensityConverter.convert(value, inputB.getUnit(),
400                         outputB.getUnit()), 0.0);
401     }
