3   import java.io.ByteArrayInputStream;
4   import java.io.ByteArrayOutputStream;
5   import java.io.IOException;
6   import java.io.ObjectInputStream;
7   import java.io.ObjectOutputStream;
8   import java.io.Serializable;
9   
10  public class SerializationHelper {
11  
12      public static <T extends Serializable> byte[] serialize(final T object) throws IOException {
13          try (final var byteArrayOutputStream = new ByteArrayOutputStream()) {
14              try (final var objectOutputStream = new ObjectOutputStream(byteArrayOutputStream)) {
15                  objectOutputStream.writeObject(object);
16              }
17  
18              byteArrayOutputStream.flush();
19  
20              return byteArrayOutputStream.toByteArray();
21          }
22      }
23  
24      public static <T extends Serializable> T deserialize(final byte[] bytes)
25              throws IOException, ClassNotFoundException {
26          try (final var byteArrayInputStream = new ByteArrayInputStream(bytes)) {
27              try (final var objectInputStream = new ObjectInputStream(byteArrayInputStream)) {
28                  //noinspection unchecked
29                  return (T) objectInputStream.readObject();
30              }
31          }
32      }
33  }
