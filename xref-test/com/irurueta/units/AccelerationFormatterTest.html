32      public void testConstructor() {
33          // test empty constructor
34          AccelerationFormatter formatter = new AccelerationFormatter();
35  
36          // check
37          assertEquals(formatter.getLocale(), Locale.getDefault());
38          assertEquals(formatter.getMaximumFractionDigits(),
39                  NumberFormat.getInstance().getMaximumFractionDigits());
40          assertEquals(formatter.getMaximumIntegerDigits(),
41                  NumberFormat.getInstance().getMaximumIntegerDigits());
42          assertEquals(formatter.getMinimumFractionDigits(),
43                  NumberFormat.getInstance().getMinimumFractionDigits());
44          assertEquals(formatter.getMinimumIntegerDigits(),
45                  NumberFormat.getInstance().getMinimumIntegerDigits());
46          assertEquals(formatter.getRoundingMode(),
47                  NumberFormat.getInstance().getRoundingMode());
48          assertEquals(formatter.getUnitSystem(), UnitLocale.getDefault());
49          assertEquals(formatter.isGroupingUsed(),
50                  NumberFormat.getInstance().isGroupingUsed());
51          assertEquals(formatter.isParseIntegerOnly(),
52                  NumberFormat.getInstance().isParseIntegerOnly());
53          assertEquals(formatter.getValueAndUnitFormatPattern(),
54                  MeasureFormatter.DEFAULT_VALUE_AND_UNIT_FORMAT_PATTERN);
55  
56          // test constructor with locale
57          final Locale locale = new Locale("es", "ES");
58          formatter = new AccelerationFormatter(locale);
59  
60          // check
61          assertEquals(formatter.getLocale(), locale);
62          assertEquals(formatter.getMaximumFractionDigits(),
63                  NumberFormat.getInstance(locale).getMaximumFractionDigits());
64          assertEquals(formatter.getMaximumIntegerDigits(),
65                  NumberFormat.getInstance(locale).getMaximumIntegerDigits());
66          assertEquals(formatter.getMinimumFractionDigits(),
67                  NumberFormat.getInstance(locale).getMinimumFractionDigits());
68          assertEquals(formatter.getMinimumIntegerDigits(),
69                  NumberFormat.getInstance(locale).getMinimumIntegerDigits());
70          assertEquals(formatter.getRoundingMode(),
71                  NumberFormat.getInstance(locale).getRoundingMode());
72          assertEquals(formatter.getUnitSystem(), UnitLocale.getFrom(locale));
73          assertEquals(formatter.isGroupingUsed(),
74                  NumberFormat.getInstance(locale).isGroupingUsed());
75          assertEquals(formatter.isParseIntegerOnly(),
76                  NumberFormat.getInstance(locale).isParseIntegerOnly());
77          assertEquals(formatter.getValueAndUnitFormatPattern(),
78                  MeasureFormatter.DEFAULT_VALUE_AND_UNIT_FORMAT_PATTERN);
79  
80          // force IllegalArgumentException
81          formatter = null;
82          try {
83              formatter = new AccelerationFormatter((Locale) null);
84              fail("IllegalArgumentException expected but not thrown");
85          } catch (final IllegalArgumentException ignore) {
86          }
87          //noinspection ConstantConditions
88          assertNull(formatter);
89  
90  
91          // test copy constructor
92          formatter = new AccelerationFormatter(locale);
93          final AccelerationFormatter formatter2 = new AccelerationFormatter(formatter);
94  
95          // check
96          assertEquals(formatter2.getLocale(), locale);
97          assertEquals(formatter2.getMaximumFractionDigits(),
98                  NumberFormat.getInstance(locale).getMaximumFractionDigits());
99          assertEquals(formatter2.getMaximumIntegerDigits(),
100                 NumberFormat.getInstance(locale).getMaximumIntegerDigits());
101         assertEquals(formatter2.getMinimumFractionDigits(),
102                 NumberFormat.getInstance(locale).getMinimumFractionDigits());
103         assertEquals(formatter2.getMinimumIntegerDigits(),
104                 NumberFormat.getInstance(locale).getMinimumIntegerDigits());
105         assertEquals(formatter2.getRoundingMode(),
106                 NumberFormat.getInstance(locale).getRoundingMode());
107         assertEquals(formatter2.getUnitSystem(), UnitLocale.getFrom(locale));
108         assertEquals(formatter2.isGroupingUsed(),
109                 NumberFormat.getInstance(locale).isGroupingUsed());
110         assertEquals(formatter2.isParseIntegerOnly(),
111                 NumberFormat.getInstance(locale).isParseIntegerOnly());
112         assertEquals(formatter2.getValueAndUnitFormatPattern(),
113                 MeasureFormatter.DEFAULT_VALUE_AND_UNIT_FORMAT_PATTERN);
114 
115         formatter = null;
116         try {
117             //noinspection ConstantConditions
118             formatter = new AccelerationFormatter((AccelerationFormatter) null);

120         } catch (final NullPointerException ignore) {
